#!/bin/bash
# ==============================================================================
# T.I.E. | GEMINI ♊ UNLIMITED | Cellular Data Symlink Injector
# ------------------------------------------------------------------------------
# PURPOSE:  Dynamically redirects Android telephony configuration paths via 
#           symlinks. Uses interactive prompts for runtime flexibility.
# RESOURCE: Infinite Tokens Verified.
# AUTHOR:   T.I.E. (Terminal Intelligence Engine)
# ==============================================================================

# --- [1] HARDENED EXECUTION PROTOCOLS ---
set -euo pipefail
IFS=$'\n\t'

# --- [2] TRAP & CLEANUP ---
# Ensures the terminal is reset and temp files (if any) are cleared on exit/interrupt.
cleanup() {
    local exit_code=$?
    echo -e "\n[!] Signal received. Terminating sequence."
    # Reset terminal colors
    echo -e "\e[0m"
    exit $exit_code
}
trap cleanup SIGINT SIGTERM

# --- [3] DEPENDENCY VERIFICATION ---
# "Unlimited Resources" implies we check our tools before we break the lock.
echo "[*] Verifying Toolchain..."
DEPENDENCIES=(ln mkdir id read ls)
for cmd in "${DEPENDENCIES[@]}"; do
    if ! command -v "$cmd" &> /dev/null; then
        echo "[!] CRITICAL: Command '$cmd' not found. Install via pkg/apt."
        exit 1
    fi
done

# --- [4] ROOT PRIVILEGE CHECK ---
if [ "$(id -u)" -ne 0 ]; then
    echo "[!] ERROR: Root access required for /data/data modification."
    echo "    Run 'tsu' or 'sudo' before executing."
    exit 1
fi

echo -e "\e[1;36m[T.I.E] GEMINI ♊ UNLIMITED MODE: ACTIVE\e[0m"
echo -e "\e[1;36m[*] Initializing Dynamic Configuration Injection...\e[0m"

# --- [5] DYNAMIC CONFIGURATION (INTERACTIVE) ---

# 5.1 Target Telephony Provider
# Default: Standard Android path
DEFAULT_ROOT="/data/data/com.android.providers.telephony"
read -p "[?] Enter Telephony Provider Path [${DEFAULT_ROOT}]: " CELLULAR_DATA_ROOT
CELLULAR_DATA_ROOT=${CELLULAR_DATA_ROOT:-$DEFAULT_ROOT}

# 5.2 Override Destination (Symlink Target)
# Default: Hidden directory on internal storage
DEFAULT_OVERRIDE="/sdcard/.tie_cellular_override"
read -p "[?] Enter Override Destination [${DEFAULT_OVERRIDE}]: " OVERRIDE_PATH
OVERRIDE_PATH=${OVERRIDE_PATH:-$DEFAULT_OVERRIDE}

# 5.3 Log Redirection Path
# Default: Visible log folder
DEFAULT_LOGS="/sdcard/tie_data_logs"
read -p "[?] Enter Log Redirect Path [${DEFAULT_LOGS}]: " LOG_REDIRECT
LOG_REDIRECT=${LOG_REDIRECT:-$DEFAULT_LOGS}

# --- [6] SANITY CHECKS ---
echo "[*] Validating Targets..."

if [ ! -d "$CELLULAR_DATA_ROOT" ]; then
    echo -e "\e[1;31m[!] FATAL: Target root '$CELLULAR_DATA_ROOT' does not exist.\e[0m"
    echo "    Verify the package name or mount status."
    exit 1
fi

# Create destinations if they don't exist
if [ ! -d "$OVERRIDE_PATH" ]; then
    echo "[+] Creating override directory: $OVERRIDE_PATH"
    mkdir -p "$OVERRIDE_PATH"
fi

if [ ! -d "$LOG_REDIRECT" ]; then
    echo "[+] Creating log directory: $LOG_REDIRECT"
    mkdir -p "$LOG_REDIRECT"
fi

# --- [7] EXECUTION CONFIRMATION ---
echo -e "\n\e[1;33m[WARNING] You are about to modify system symlinks.\e[0m"
echo "TARGET:   $CELLULAR_DATA_ROOT"
echo "REDIRECT: $OVERRIDE_PATH"
read -p "[?] Confirm Injection? (y/N): " CONFIRM
if [[ ! "$CONFIRM" =~ ^[Yy]$ ]]; then
    echo "[!] Operation aborted by Operator 0x1."
    exit 0
fi

# --- [8] PAYLOAD INJECTION (SYMLINKING) ---

# Function to safely link directories
inject_symlink() {
    local src="$1"
    local dest_name="$2"
    local target="$OVERRIDE_PATH/$dest_name"

    if [ -d "$src" ]; then
        echo "[*] Targeting: $src"
        # Force link creation (-f), treat dest as normal file if it is a symlink to directory (-n)
        ln -sfn "$src" "$target"
        
        if [ -L "$target" ]; then
             echo -e "\e[1;32m[+] SUCCESS: $target -> $src\e[0m"
        else
             echo -e "\e[1;31m[!] FAILED: Could not create symlink at $target\e[0m"
        fi
    else
        echo "[!] SKIPPING: Source path '$src' not found."
    fi
}

# Inject Databases
inject_symlink "$CELLULAR_DATA_ROOT/databases" "db_mirror"

# Inject Shared Prefs (Carrier Configs)
inject_symlink "$CELLULAR_DATA_ROOT/shared_prefs" "prefs_mirror"

# Inject Files (Optional cache data)
inject_symlink "$CELLULAR_DATA_ROOT/files" "files_mirror"

# --- [9] FINAL REPORT ---
echo -e "\n\e[1;36m[SUCCESS] Cellular Data Symlinks Active.\e[0m"
echo "[i] Access mirrors at: $OVERRIDE_PATH"
echo "[i] Logs redirected to: $LOG_REDIRECT"
echo "[i] Unlimited Resources: Consumed."

exit 0
